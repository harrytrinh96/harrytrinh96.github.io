## The time complexity is O(N) where N is the length of the given integer
## The space complexity is O(1) because the solution does not save anything in memory.
def solution(N):
    # write your code in Python 3.6
    ## dealing with a non-negative integer
    if N >=  0:
        ## first, convert N to String
        num = str(N)
        # loop according to the length of num
        for i in range(len(num)):
            if '5' > num[i]:
                return int(num[:i] + '5' + num[i:])

    ## dealing with a negative integer
    else:
        ## first, convert N to String
        num = str(N)
        # loop according to the length of num
        for i in range(1, len(num)): # starts from 1 because index 0 is "-"
            if '5' < num[i]:
                return int(num[:i] + '5' + num[i:])
    return int(num + ' 5')

print(solution(268))

print(solution(-999))
